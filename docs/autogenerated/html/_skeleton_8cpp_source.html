62Skeleton::Skeleton(std::filesystem::path path)
63{
64 std::filesystem::path skl_filename = path / "skeleton.skl";
65 std::filesystem::path rest_filename = path / "rest.anmr";
66 // collecting rest transforms
67 std::ifstream file(rest_filename);
68 if (!file)
69 throw std::logic_error(std::string("couldn't find .anmr file: ") + rest_filename.string());
70 std::string line;
71 while (getline(file, line))
72 rest_transforms.push_back(str_to_transform(line));
73 file.close();
74
75 // loading bones
76 std::vector<Skeleton::Bone> bones;
77 file.open(skl_filename);
78 if (!file)
79 throw std::logic_error(std::string("couldn't find .skl file: ") + skl_filename.string());
80 size_t i = 0;
81 while (getline(file, line))
82 {
83 Bone bone;
84 bone.index = i;
85 std::vector<std::string> parts = Engine::split(line, ' ');
86 bone.name = parts[0];
87 bone.parent_name = parts[1];
88 bone.length = stof(parts[2]);
89 bone.parent = nullptr; // this should be null by default
90 if (bone.parent_name.empty())
91 root_bones.push_back(bone);
92 else
93 bones.push_back(bone);
94 i++;
95 }
96 file.close();
97
98 // creating the tree of bones
99 while (!bones.empty())
100 {
101 for (auto it = bones.begin(); it != bones.end(); it++)
102 {
103 Bone* bone = Bone::search_in_bones(it->parent_name, root_bones);
104 if (bone)
105 {
106 it->parent = bone;
107 bone->children.push_back(*it);
108 it = bones.erase(it) - 1;
109 }
110 }
111 }
112
113 // bones to rest_matrices and to bone_transforms
114 size_t count = Bone::count_of_bones(root_bones);
115 rest_matrices.resize(count);
116 bone_transforms.resize(count);
117 Bone::on_bones(root_bones, [&](Bone* bone){
118 bone_transforms[bone->index] = rest_transforms[bone->index]; // local to object transform, not to parent bone
119 rest_matrices[bone->index] = bone_transforms[bone->index].get_matrix();
120 });
121 models.resize(Bone::count_of_bones(root_bones), glm::mat4(1.0f));
122}

124void Skeleton::add_animation(std::filesystem::path path)
125{
126 std::ifstream file(path);
127 if (!file)
128 throw std::logic_error(std::string("couldn't find .anm file: ") + path.string());
129 std::string line;
130 // getting data from file
131 // fps
132 getline(file, line);
133 auto vec = Engine::split(line, '#');
134 float fps = stof(vec[0]);
135 // frames
136 getline(file, line);
137 vec = Engine::split(line, '#');
138 int frames = stoi(vec[0]);
139 // sample rate
140 getline(file, line);
141 vec = Engine::split(line, '#');
142 int sample_rate = stoi(vec[0]);
143 // bones count
144 getline(file, line);
145 vec = Engine::split(line, '#');
146 int bone_count = stoi(vec[0]);
147 if (static_cast<size_t>(bone_count) != Bone::count_of_bones(root_bones))
148 throw std::logic_error(std::string("number of bones of animation is not equal to bones of skeleton, in file: ") + path.string());
149
150 Animation* animation = new Animation;
151 animation->duration = frames * sample_rate / fps;
152 animation->frames.resize(frames);
153 size_t i = 0;
154 while (getline(file, line))
155 {
156 animation->frames[i / bone_count].push_back(Skeleton::str_to_transform(line));
157 i++;
158 }
159 file.close();
160 //std::cout << "frames of animation: " << animation->frames.size() << std::endl;
161 //if (!animation->frames.empty())
162 // std::cout << "bones per frame: " << animation->frames[0].size() << std::endl;
163 animations[path.stem().string()].reset(animation);
164}

173Engine::span<Engine::anim_mat4> Skeleton::get_animation_matrices(AnimationPlayer& player, bool advance_time)
174{
175 // Getting and updating bone_transforms
176 auto& animation = player.animations[0];
177 if (!animations.count(animation.name))
178 throw std::logic_error(std::string("couldn't find the animation named: ") + animation.name);
179 auto& anim = animations[animation.name];
180 auto& frame = anim->frames[static_cast<int>(fmod(player.time, anim->duration) * anim->frames.size() / anim->duration)];
181 bone_transforms = frame; // copying frame to bones
182
183 // Getting matrices from bone_transforms
184 auto it = bone_transforms.begin();
185 auto jt = rest_matrices.begin();
186 auto kt = models.begin();
187 for (auto ft = bone_transforms.end(); it != ft; it++, jt++, kt++)
188 (*kt) = it->get_matrix() * glm::inverse(*jt);
189 if (advance_time)
190 player.time += App::app->delta_time;
191 return Engine::span<Engine::anim_mat4>(models.data(), models.size());
192}
