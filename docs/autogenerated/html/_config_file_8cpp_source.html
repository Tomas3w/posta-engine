85ConfigFile::ConfigFile(std::filesystem::path path)
86{
87 std::ifstream file(path);
88 if (!file)
89 throw std::logic_error(std::string("Couldn't load file '") + path.string() + "'");
90 std::string line;
91 size_t line_pos;
92 while (getline(file, line))
93 {
94 // Skipping comments
95 size_t comment_pos = line.find('#');
96 if (comment_pos != std::string::npos)
97 line.resize(comment_pos);
98 // Skipping lines that only contain whitespaces
99 bool is_other_than_whitespace = false;
100 for (char& c : line)
101 {
102 if (!std::isspace(c))
103 {
104 is_other_than_whitespace = true;
105 break;
106 }
107 }
108 if (!is_other_than_whitespace)
109 continue;
110
111 // Dividing the string in two parts
112 size_t pos = line.find('=');
113 if (pos == std::string::npos)
114 throw std::logic_error(std::string("Couldn't find the '=' in one of the config pairs, in line ") + std::to_string(line_pos + 1));
115 std::string left, right;
116 left.insert(left.begin(), line.begin(), line.begin() + pos);
117 right.insert(right.begin(), line.begin() + pos + 1, line.end());
118
119 // Adding this to the raw data
120 data[Engine::strip(left)] = Engine::strip(right);
121 line_pos++;
122 }
123 file.close();
124}
