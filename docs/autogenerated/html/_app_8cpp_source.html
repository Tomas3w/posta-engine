162void App::loop()
163{
164 Uint32 ticks;
165 SDL_Event event;
166
167 while (!exit_loop)
168 {
169 ticks = SDL_GetTicks();
170 clear_frame_and_flip();
171
172
173 mouse_state = SDL_GetMouseState(&mouse_x, &mouse_y);
174 mouse_y = get_height() - mouse_y;
175 on_frame(); // runs custom code for each frame
176 on_draw();
177 disable_depth_test();
178 shader2d->bind();
179 on_draw_2d();
180 enable_depth_test();
181
182 if (camera)
183 camera->loop();
184 while (SDL_PollEvent(&event))
185 {
186 switch (event.type)
187 {
188 case SDL_WINDOWEVENT:
189 switch (event.window.event)
190 {
191 case SDL_WINDOWEVENT_RESIZED:
192 this->width = event.window.data1;
193 this->height = event.window.data2;
194 glViewport(0, 0, width, height);
195
196 // updates default camera projection matrix.
197 if (camera)
198 camera->set_resolution(width, height);
199 break;
200 }
201 break;
202 case SDL_QUIT:
203 exit_loop = true;
204 }
205 on_event(event);
206 }
207 step_physics();
208
209 // Updates delta time between frames
210 Uint32 nticks = SDL_GetTicks();
211 delta_time = (nticks - ticks) / 1000.0f;
212
213 // corrects framerate
214 const float framerate = 1 / 60.0f;
215 if (delta_time < framerate)
216 {
217 float diff = framerate - delta_time;
218 while ((SDL_GetTicks() - nticks) / 1000.0f < diff);
219 delta_time = (SDL_GetTicks() - ticks) / 1000.0f;
220 }
221 }
222
223}
